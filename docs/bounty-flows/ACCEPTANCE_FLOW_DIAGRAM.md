```
          expanded={wipExpanded}
          onToggle={toggleWip}
        >
          {isOwner ? (
            <>
              <MessageBar 
                conversationId={conversation.id}
                onSendMessage={handleSendMessage}
              />
              <AttachmentsList 
                attachments={attachments}
              />
              <RatingStars 
                rating={ratingDraft}
                onRatingChange={setRatingDraft}
              />
            </>
          ) : (
            <>
              <InfoBox>
                "Begin work on the bounty..."
              </InfoBox>
              <AttachmentsList 
                attachments={attachments}
              />
              <NextButton onPress={goToReview} />
            </>
          )}
        </AnimatedSection>
      )}
      
      <QuickActions>
        {actionButtons}
      </QuickActions>
    </View>
  )}
</MyPostingExpandable>
```

## ğŸ¯ Key Decisions & Trade-offs

### Decision 1: Delete vs Mark Rejected
**Choice:** Delete competing requests  
**Why:** Cleaner state, no orphaned data, hunters get clear feedback  
**Trade-off:** Can't see rejection history without separate audit log

### Decision 2: Component Composition
**Choice:** Small, focused components  
**Why:** Reusability, testability, easier to maintain  
**Trade-off:** More files, slightly more complex imports

### Decision 3: Local Rating Draft
**Choice:** Store in component state, not persisted  
**Why:** Simpler implementation, draft nature  
**Trade-off:** Lost on unmount, but acceptable for drafts

### Decision 4: Optimistic Updates
**Choice:** Update UI immediately, rollback on error  
**Why:** Better UX, feels instant  
**Trade-off:** Need careful error handling and state management

### Decision 5: Promise.all for Cleanup
**Choice:** Delete competing requests in parallel  
**Why:** Faster, more efficient  
**Trade-off:** If one fails, others succeed (acceptable)
